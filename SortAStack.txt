void sortedInsert(stack<int> &stack, int num)
{
	// base case
	if(stack.empty() || stack.top() < num)
	{
		stack.push(num);
		return;
	}

	int n=stack.top();
	stack.pop();

	// Recursive call
	sortedInsert(stack, num);
	stack.push(n);
}

void sortStack(stack<int> &stack)
{
	// Write your code here

	// base case
	if(stack.empty())
	{
		return;
	}

	int num=stack.top();
	stack.pop();

	// Recursive Call
	sortStack(stack);

	sortedInsert(stack,num);
}

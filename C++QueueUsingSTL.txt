            		// C++ Queue Using STL //

#include <bits/stdc++.h>
using namespace std;

int main() 
{
    // create a queue
    queue<int> q;
    
    q.push(11);
    cout<<"Front element: "<<q.front()<<endl;
    
    q.push(12);
    cout<<"Front element: "<<q.front()<<endl;
    
    q.push(13);
    cout<<"Front element: "<<q.front()<<endl;
   
    q.pop();
    q.pop();
    
    cout<<"size of queue: "<<q.size()<<endl;
    
    if(q.empty())
    {
        cout<<"Queue is empty"<<endl;
    }
    else
    {
        cout<<"Queue is not empty"<<endl;
    }
}
